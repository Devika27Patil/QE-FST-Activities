-- Activity 9 --
1.
SELECT 
    s.salesman_name,
    c.customer_name
FROM salesman s
JOIN customers c
ON s.salesman_id = c.salesman_id;

2.
SELECT 
    c.customer_name,
    s.salesman_name,
    c.grade
FROM customers c
JOIN salesman s
ON c.salesman_id = s.salesman_id
WHERE c.grade < 300
ORDER BY c.customer_name ASC;

3.
SELECT
    c.customer_name,
    s.salesman_name,
    s.commission
FROM customers c
JOIN salesman s
ON c.salesman_id = s.salesman_id
WHERE s.commission > 12;

4.
SELECT 
    o.order_no,
    o.order_date,
    o.purchase_amount,
    c.customer_name,
    s.salesman_name,
    s.commission
FROM orders o
JOIN CUSTOMERS c
ON o.customer_id = c.customer_id
Join salesman s
ON o.salesman_id = s.salesman_id;
